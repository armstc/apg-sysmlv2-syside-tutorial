package GarageDoorOpenerModel {
    private import ScalarValues::*;
    private import USCustomaryUnits::*;

    package Requirements {
        requirement Functionality {
            requirement <'REQ-01'> openGarageDoor {
                doc
                /* The system shall raise the garage door when the user presses the door control button. */
            }
            requirement <'REQ-02'> closeGarageDoor;
            requirement <'REQ-03'> illuminateGarage;
        }
        requirement Integration {
            requirement supportRemoteControlOpeners;
        }
        requirement Safety {
            requirement manualOpening;
            requirement preventCrushing;
            requirement preventEntrapment;
        }
        requirement Performance {
            requirement maximumDoorWeight {
                doc
                /* The system shall be able to raise and lower a garage door that weighs up to 350 pounds. */
                attribute maxDoorWeight :> ISQ::mass;
                require constraint { maxDoorWeight <= 350 [lb] }
            }
        }
    } // package Requirements

    package Behavior {
        attribute doorFullyOpened : Boolean = false;

        action def OpenGarageDoor {
            first start;
            then requestOpen;

            action requestOpen;
            then f1;

            fork f1;
                then raiseDoor;
                then monitorObstructions;

            action raiseDoor;
            then j1;

            action monitorObstructions;
            then j1;

            join j1;
            then d1;
            decide d1;
                if doorFullyOpened then indicateFullyOpened;
                if not doorFullyOpened then indicatePartiallyOpened;

            merge m1;
            then done;

            action indicateFullyOpened;
            then m1;

            action indicatePartiallyOpened;
            then m1;
        }
    } // package Behavior
    package LogicalStructure {
        part def GarageDoorOpener {
            perform openGarageDoor.indicateFullyOpened;
            perform openGarageDoor.indicatePartiallyOpened;
        }

        part def GarageDoor {}

        part def DoorControl {
            perform openGarageDoor.requestOpen;
        }

        part def Actuator {
            perform openGarageDoor.raiseDoor;
        }

        part def ObstructionSensor {
            perform openGarageDoor.monitorObstructions;
        }

        part def GarageDoorOpenerSystem {
            ref part door : GarageDoor;
            part opener : GarageDoorOpener;
            part actuator : Actuator;
            part control : DoorControl;
            part sensor : ObstructionSensor;

            connect opener to actuator;
            connect actuator to door;
            connect opener to control;
            connect opener to sensor;
        }
    } // package LogicalStructure
    action openGarageDoor : Behavior::OpenGarageDoor;
} // package GarageDoorOpenerModel
